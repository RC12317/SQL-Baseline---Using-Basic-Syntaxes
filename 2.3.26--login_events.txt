SELECT profileName,
       timestamp,
       COUNT(DISTINCT ip) AS IP,
       COUNT(DISTINCT email) AS email
FROM login_events
GROUP BY hour_utc
ORDER BY IP ASC;


SELECT profileName,
       timestamp,
       COUNT(DISTINCT ip) AS IP,
       COUNT(DISTINCT email) AS email
FROM login_events
GROUP BY userAgent
ORDER BY IP ASC;
